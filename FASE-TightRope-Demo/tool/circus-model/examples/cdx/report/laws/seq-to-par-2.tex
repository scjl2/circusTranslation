  A_1 \circseq A_2 ~ \equiv ~ ((A_1 \circseq c~!~x \then \Skip) \lpar \wrt(A_1) | \lchanset c \rchanset | \wrt(A_2) \rpar (c~?~x \then A_2)) \circhide \lchanset c \rchanset

\end{circusaction*}
